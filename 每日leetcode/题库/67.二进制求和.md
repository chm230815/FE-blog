/**
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
var addBinary = function(a, b) {
  const len = Math.max(a.length, b.length);
  let curA = []
  let curB = []
  for(let i = 0; i < len; i++) {
    if(a[i]){
      curA.push(a[i])
    } else {
      curA.unshift(0)
    }
    if(b[i]){
      curB.push(b[i])
    }
    else {
      curB.unshift(0)
    }
  }
  let num = 0
  let resList = []
  for(let i = len - 1; i >= 0; i--) {
    const numA = Number(curA[i])
    const numB = Number(curB[i])
    if(numA + numB + num >= 2){
      resList[i] = (numA + numB + num) % 2
      num = 1
    } else {
      resList[i] = numA + numB + num
      num = 0
    }
  }
  if(num) {
    resList.unshift(1)
  }
  // 数组转字符串
  const res = resList.join('')
  return res
};
