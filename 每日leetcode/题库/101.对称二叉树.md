```md
示例 2：
```

```js
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isSymmetric = function(root) {
  let arrA = []
  let arrB = []
  const fn1 = (root) => {
    if(root){
      arrA.push(root.val)
      fn1(root.left)
      fn1(root.right)
    } else {
      arrA.push(null)
    }
  }
  const fn2 = (root) => {
    if(root){
      arrB.push(root.val)
      fn2(root.right)
      fn2(root.left)
    } else {
      arrB.push(null)
    }
  }
  fn1(root.left)
  fn2(root.right)
  if(arrA.length !== arrB.length) return false
  for(let i = 0; i < arrA.length; i++){
    if(arrA[i] !== arrB[i]) return false
  }
  return true
};
```
