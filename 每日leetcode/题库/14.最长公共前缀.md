/**
 * @param {string[]} strs
 * @return {string}
 */
var longestCommonPrefix = function(strs) {
  let minLength = strs[0].length
  strs.forEach(item => {
    if(minLength > item.length){
      minLength = item.length
    }
  })
  let resStr = ''
  for(let i = 0; i < minLength; i++){
    let str = strs[0][i]
    for(let j = 0; j < strs.length; j++){
      if(strs[j][i] !== str) return resStr
    }
    resStr += str
  }
  return resStr
};
