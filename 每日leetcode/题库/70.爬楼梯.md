// 排列组合找规律发现 这属于什么那契科夫什么的数列 
// 1 1
// 2 2
// 3 3
// 4 5
// 5 8
// 6 13
// 7 21
var climbStairs = function(n) {
  if(n <= 2) return n
  let num1 = 1
  let num2 = 2
  let sumNum = 2
  let res = 0
  while(sumNum < n){
    res = num1 + num2
    num1 = num2
    num2 = res
    sumNum ++ 
  }
  return res
};
