/**
 * @param {number} x
 * @return {boolean}
 */
var isPalindrome = function(x) {
  let y = ''
  let strx = x.toString()
  const length = strx.length
  for(let i = length-1; i >= 0; i--){
    y += strx[i]
  }
  if(y == strx){
    return true
  }else{
    return false
  }
};
